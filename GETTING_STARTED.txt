STEP 4: CREATE YOUR OWN PLOT
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Create my_plot.cpp:

    #include <perspicillum.hpp>
    #include <vector>
    #include <cmath>

    int main() {
      std::vector<double> x, y;
      for (double t = 0; t < 6.28; t += 0.1) {
        x.push_back(t);
        y.push_back(std::sin(t));
      }

      perspicillum::plot(x, y)
        .title("My First Plot")
        .name("sin(x)")
        .color(0.2f, 0.6f, 1.0f)
        .show();
    }

Compile:
  g++ -std=c++20 my_plot.cpp \
    -L /home/irz/Documents/perspicillum/build \
    -I /home/irz/Documents/perspicillum/include \
    -l perspicillum -l glfw -l GL -o my_plot

Run:
  ./my_plot

STEP 5: USE IN YOUR PROJECT
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Add to your CMakeLists.txt:

    # At the top with other add_subdirectory calls
    add_subdirectory(perspicillum)

    # In your executable/library
    add_executable(my_app main.cpp)
    target_link_libraries(my_app perspicillum)
    set_property(TARGET my_app PROPERTY CXX_STANDARD 20)

Then in your code:

    #include <perspicillum.hpp>

    int main() {
      perspicillum::plot(x, y).show();
    }

Example 1: Express API (Quick Plot)

  #include <perspicillum.hpp>

  int main() {
    std::vector<double> x = {1, 2, 3, 4, 5};
    std::vector<double> y = {1, 4, 9, 16, 25};
    
    perspicillum::plot(x, y).show();
  }

Example 5: Multiple Subplots
────────────────────────────

  perspicillum::Figure fig(2, 2);

  fig.subplot(1, 1).add_plot(
    std::make_unique<perspicillum::LinePlot>(
      perspicillum::make_data(x1),
      perspicillum::make_data(y1)
    )
  ).style().set_color(1.0f, 0.0f, 0.0f);

  fig.subplot(1, 2).add_plot(
    std::make_unique<perspicillum::ScatterPlot>(

  fig.show();

Example 6: Eigen Vectors
─────────────────────────

  #include <Eigen/Dense>
  #include <perspicillum.hpp>

  int main() {
    Eigen::VectorXd x = Eigen::VectorXd::LinSpaced(50, 0, 2*M_PI);
    Eigen::VectorXd y = x.array().sin();

    perspicillum::plot(x, y)
      .title("Eigen Vector Plot")
      .show();
  }

COMMON METHODS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

PlotExpression (return from plot/scatter/bar):
  .title(string)              Set window title
  .name(string)               Set legend name
  .color(r, g, b, a)          Set color (0-1 range, a optional)
  .width(float)               Set line width in pixels
  .markersize(int)            Set marker size in pixels
  .show()                     Display in window

Figure:
  Figure()                    Single subplot
  Figure(rows, cols)          Create grid
  .subplot(r, c)              Get subplot (1-indexed)
  .add_line(x, y)             Add line plot
  .add_scatter(x, y)          Add scatter plot
  .add_bar(x, y)              Add bar plot
  .set_title(title)           Set figure title
  .set_size(w, h)             Set window size
  .show()                     Display
  .render()                   Render (no window)

PlotStyle (result of .style()):
  .set_color(r, g, b, a)      Set color
  .set_name(label)            Set label
  .set_width(w)               Set line width
  .set_markersize(s)          Set marker size
  .set_showlegend(bool)       Show in legend

TROUBLESHOOTING
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Build fails:
  ✓ Ensure C++20: cmake .. -DCMAKE_CXX_STANDARD=20
  ✓ Install deps: apt-get install libglfw3-dev libgl1-mesa-dev libeigen3-dev
  ✓ Clean build: rm -rf build && mkdir build && cd build

Compilation errors:
  ✓ Use -std=c++20 flag
  ✓ Include perspicillum.hpp, not individual headers
  ✓ Check all libraries are linked: -lglfw -lGL -lm

Plot doesn't display:
  ✓ Check X11 is available (DISPLAY env var set)
  ✓ Verify OpenGL: glxinfo
  ✓ Call .show() at end
  ✓ Check data isn't empty or NaN

Integration issues:
  ✓ Link library: target_link_libraries(app perspicillum)
  ✓ Add include: target_include_directories(app PRIVATE perspicillum/include)
  ✓ Set C++20: set_property(TARGET app PROPERTY CXX_STANDARD 20)

Performance:
  ✓ Use std::span for zero-copy access
  ✓ Pass large vectors with std::move
  ✓ Build in Release: cmake .. -DCMAKE_BUILD_TYPE=Release
